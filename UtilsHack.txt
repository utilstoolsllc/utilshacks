-- Key Okuma Fonksiyonu
local function readKey() 
    if isfile(keyFile) then 
        local content = readfile(keyFile)
        local success, decoded = pcall(function() return HttpService:JSONDecode(content) end)
        if success and decoded.key then return decoded.key end 
    end 
    return nil 
end

-- Script Sonu Splash ve Bilgi Güncelleme
local function setBranding()
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    -- RE yollarının oyununuza göre doğru olduğundan emin olun
    local remoteEvent = ReplicatedStorage:FindFirstChild("RE") and ReplicatedStorage.RE:FindFirstChild("1RPNam1eTex1t")
    local colorEvent = ReplicatedStorage:FindFirstChild("RE") and ReplicatedStorage.RE:FindFirstChild("1RPNam1eColo1r")
    
    if remoteEvent then
        remoteEvent:FireServer("RolePlayBio", "Created by UtilsTools")
        remoteEvent:FireServer("RolePlayName", "[ UTILSTOOLS HUB V4 - USER ]")
        if colorEvent then
            colorEvent:FireServer("PickingRPBioColor", Color3.fromRGB(164, 18, 0))
        end
        
        wait(5)
        
        remoteEvent:FireServer("RolePlayBio", "")
        remoteEvent:FireServer("RolePlayName", "")
    end
end
